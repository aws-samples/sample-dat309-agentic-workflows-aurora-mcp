from strands import Agent, tool
from strands.models import BedrockModel
from lib.aurora import run_sql

@tool
def search_packages(category: str,
                    max_price: float) -> list:
    """Filter trip_packages by category + price."""
    return run_sql(
      "SELECT * FROM trip_packages "
      "WHERE category = :c AND price <= :p",
      c=category, p=max_price)

agent = Agent(
    model=BedrockModel("claude-sonnet"),
    tools=[search_packages, check_dates, book_trip],
)
agent("city breaks under $1500")
